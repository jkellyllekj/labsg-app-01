async function cycleBackgroundManually() {
  const btn = document.getElementById("bgCycleBtn");
  const bgA = document.getElementById("bgA");
  const bgB = document.getElementById("bgB");
  if (!btn || !bgA || !bgB) return;

  if (btn.disabled) return;
  btn.disabled = true;

  const nextIndex = (bgIndex + 1) % backgroundImages.length;
  const nextUrl = backgroundImages[nextIndex];

  // Preload BEFORE changing any opacity/classes
  try {
    await preloadImage(nextUrl);
  } catch (e) {
    btn.disabled = false;
    return;
  }

  const fromLayer = activeBgLayer === "A" ? bgA : bgB;
  const toLayer   = activeBgLayer === "A" ? bgB : bgA;

  // Prepare next layer hidden with new image
  toLayer.classList.remove("isActive");           // ensure opacity 0 baseline
  toLayer.style.backgroundImage = `url(${nextUrl})`;

  // Force the browser to register the baseline state so the transition triggers
  void toLayer.offsetHeight;

  // OVERLAP: turn the new one on first (old stays on)
  toLayer.classList.add("isActive");

  // After the fade duration, turn the old one off
  window.setTimeout(() => {
    fromLayer.classList.remove("isActive");

    bgIndex = nextIndex;
    activeBgLayer = activeBgLayer === "A" ? "B" : "A";
    btn.disabled = false;
  }, 280); // slightly > 260ms
}
